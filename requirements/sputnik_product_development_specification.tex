\begin{description}
	\item[Requirements] \hfill \\
		\textbf{Must}
		\begin{itemize}
			\item Environment
			\begin{itemize}
				\item Have a radiation tolerant watchdog system
				\item Be able to operate in an Industrial Operating Temperature Range (?40°C to 80°C)
				\item Acceleration: must handle 15g in the "Z" axis
				\item Vibration TBD
			\end{itemize}
			\item Radio System
			\begin{itemize}
				\item Operate in the 70 cm band (436 to 438 MHz)
				\item Use IARU specified frequency for RF comms and make appropriate changes if necessary
				\item Meet FCC Amateur Radio Licensing requirements (Title 47 CFR part 97)
				\item Meet 400km ISS orbit link budget by a margin of 6 dB
			\end{itemize}
			\item Energy storage
			\begin{itemize}
				\item Use an energy storage system to power the energy when not in sunlight
				\item Monitor the state of charge of the battery
				\begin{itemize}
					\item Voltage
					\item Current
					\item Charge (Coulombs)
				\end{itemize}
				\item Charge the energy storage from energy harvesting system
			\end{itemize}
			\item Energy harvesting
			\begin{itemize}
				\item Handle 6 photovoltaic panel inputs
			\end{itemize}
			\item Energy Switching/Control
			\begin{itemize}
				\item Be able to seamlessly switch energy sources to the load
			\end{itemize}
			\item System controller
			\begin{itemize}
				\item Have an interface to the CubeSat payload
				\begin{itemize}
					\item One or more standard logic level UARTs for communicating with payload
					\item Power switches to turn payload on and off
				\end{itemize}
				\item Monitor energy storage and energy harvesting systems
			\end{itemize}
			\item CubeSat requirements
			\begin{itemize}
				\item Meets latest CubeSat specification
				\item Fits in 1/4 1U CubeSat
				\item Weighs less than 250 g
			\end{itemize}
		\end{itemize}
		\textbf{Should}
		\begin{itemize}
			\item Use as many COTS (Commercial Off The Shelf) parts as possible
			\item Use Automotive Operating Temperature Range (-40C to 125C) or even military (-55C to 125C)
			\item Use a frequency ofr 436.5 MHz for RF communication
		\end{itemize}
	
		\textbf{May}
\end{description}
